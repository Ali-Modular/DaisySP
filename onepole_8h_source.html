73 inline float Process(float in)
74 {
75 float lp;
76 lp = (g_ * in + state_) * gi_;
77 state_ = g_ * (in - lp) + lp;
78
79 switch(mode_)
80 {
81 case FILTER_MODE_LOW_PASS: return lp;
82 case FILTER_MODE_HIGH_PASS: return in - lp;
83 }
84
85 return 0.0f;
86 }
